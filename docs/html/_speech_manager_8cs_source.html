251  public static byte[] ReadToEnd(Stream stream)
252  {
253  long originalPosition = 0;
254 
255  if (stream.CanSeek)
256  {
257  originalPosition = stream.Position;
258  stream.Position = 0;
259  }
260 
261  try
262  {
263  byte[] readBuffer = new byte[4096];
264 
265  int totalBytesRead = 0;
266  int bytesRead;
267 
268  while ((bytesRead = stream.Read(readBuffer, totalBytesRead, readBuffer.Length - totalBytesRead)) > 0)
269  {
270  totalBytesRead += bytesRead;
271 
272  if (totalBytesRead == readBuffer.Length)
273  {
274  int nextByte = stream.ReadByte();
275  if (nextByte != -1)
276  {
277  byte[] temp = new byte[readBuffer.Length * 2];
278  Buffer.BlockCopy(readBuffer, 0, temp, 0, readBuffer.Length);
279  Buffer.SetByte(temp, totalBytesRead, (byte)nextByte);
280  readBuffer = temp;
281  totalBytesRead++;
282  }
283  }
284  }
285 
286  byte[] buffer = readBuffer;
287  if (readBuffer.Length != totalBytesRead)
288  {
289  buffer = new byte[totalBytesRead];
290  Buffer.BlockCopy(readBuffer, 0, buffer, 0, totalBytesRead);
291  }
292  return buffer;
293  }
294  finally
295  {
296  if (stream.CanSeek)
297  {
298  stream.Position = originalPosition;
299  }
300  }

356  private static float[] AudioWithHeaderToUnityAudio(byte[] wavAudio, out int sampleCount, out int frequency)
357  {
358  // Determine if mono or stereo
359  int channelCount = wavAudio[22]; // Speech audio data is always mono but read actual header value for processing
360  Debug.Log($"Audio data has {channelCount} channel(s).");
361 
362  // Get the frequency
363  frequency = BytesToInt(wavAudio, 24);
364  Debug.Log($"Audio data frequency is {frequency}.");
365 
366  // Get past all the other sub chunks to get to the data subchunk:
367  int pos = 12; // First subchunk ID from 12 to 16
368 
369  // Keep iterating until we find the data chunk (i.e. 64 61 74 61 ...... (i.e. 100 97 116 97 in decimal))
370  while (!(wavAudio[pos] == 100 && wavAudio[pos + 1] == 97 && wavAudio[pos + 2] == 116 && wavAudio[pos + 3] == 97))
371  {
372  pos += 4;
373  int chunkSize = wavAudio[pos] + wavAudio[pos + 1] * 256 + wavAudio[pos + 2] * 65536 + wavAudio[pos + 3] * 16777216;
374  pos += 4 + chunkSize;
375  }
376  pos += 8;
377 
378  // Pos is now positioned to start of actual sound data.
379  sampleCount = (wavAudio.Length - pos) / 2; // 2 bytes per sample (16 bit sound mono)
380  if (channelCount == 2) { sampleCount /= 2; } // 4 bytes per sample (16 bit stereo)
381  Debug.Log($"Audio data contains {sampleCount} samples. Starting conversion");
382 
383  // Allocate memory (supporting left channel only)
384  var unityData = new float[sampleCount];
385 
386  try
387  {
388  // Write to double array/s:
389  int i = 0;
390  while (pos < wavAudio.Length)
391  {
392  unityData[i] = BytesToFloat(wavAudio[pos], wavAudio[pos + 1]);
393  pos += 2;
394  if (channelCount == 2)
395  {
396  pos += 2;
397  }
398  i++;
399  }
400  }
401  catch (Exception ex)
402  {
403  Debug.Log($"Error occurred converting audio data to float array of size {wavAudio.Length} at position {pos}.");
404  }
405 
406  return unityData;
407  }
